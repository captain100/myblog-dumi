      function getTime() {
        var nowTime = new Date();
        var h = nowTime.getHours();
        var m = nowTime.getMinutes();
        var s = nowTime.getSeconds();

        function checktime(x) {
          if (x < 10) {
            x = '0' + x;
          }
          return x;
        }
        h = checktime(h);
        m = checktime(m);
        s = checktime(s);
        return `${h} : ${m} : ${s}`;
      }

      var times = document.querySelectorAll('text');
      setInterval(() => {
        var nowTimer = getTime();
        times.forEach((item, i) => {
          item.innerHTML = nowTimer;
        });
      }, 1000);
